extern printf
section .data
	msg_a_:	db	"a",0
	const_:	dd	15
	const_:	dd	4
	printf_flt_:	db	"%f",10,0
	printf_int_:	db	"%i",10,0
	printf_str_:	db	"%s = ",0
	printf_arr_:	db	"%s[%i] = ",0
section .bss
	a:	resd	1
section .text
	global main
main:
	finit
	fild	dword [const_]
	fild	dword [const__]
	fsub
	fistp	dword [a]
	mov		edx, 0
	push	edx
	push	dword msg_a_
	push	dword printf_arr_
	call	printf
	add		esp, 12
	push	dword [a+0*4]
	push	dword printf_int_
	call	printf
	add		esp, 8
	mov		eax, 0
	ret
